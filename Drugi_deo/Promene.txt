CREATE OR REPLACE TRIGGER "POLICAJAC_AUTO_PK" 
    BEFORE INSERT
    ON POLICAJAC
    REFERENCING NEW AS NEW
    FOR EACH ROW
BEGIN
    SELECT POLICAJAC_SEQ.NEXTVAL INTO :NEW.ID FROM DUAL;
END;

ALTER TRIGGER "ODELJENJE_AUTO_PK" ENABLE;


CREATE SEQUENCE POLICAJAC_SEQ
